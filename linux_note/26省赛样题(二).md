<font style="background-color:rgba(255, 255, 255, 0);">[root@ls2 ~]# </font>`<font style="background-color:rgba(255, 255, 255, 0);">cat /etc/nginx/conf.d/1.conf</font>`

```plain
upstream tomcat{
        server ls3.gdskills.cn:443 max_fails=3 fail_timeout=10s;
        server ls4.gdskills.cn:443 max_fails=3 fail_timeout=10s;
}
server {
        listen 80;
        server_name tomcat.gdskills.cn;
        return 301 https://tomcat.gdskills.cn;
}
server{
        listen 443 http2 ssl;
        server_name tomcat.gdskills.cn;
        ssl_certificate "/etc/pki/tls/skills.crt";
        ssl_certificate_key "/etc/pki/tls/skills.key";
        location / {
                proxy_pass https://tomcat;
        }
}
```
