## linux基础命令
1. `cd`  ==用于切换用户所在的目录==
    > root@dell-PowerEdge-R740xd:**~**# cd /opt/  
    root@dell-PowerEdge-R740xd:**/opt**# cd /opt/gitlab/  
    root@dell-PowerEdge-R740xd:**/opt/gitlab**#  
    
2. `touch` ==新建一个空文件/将文件的访问和修改时间更新至当前==  
    > root@dell-PowerEdge-R740xd:~# touch 123.txt  #新建空文件
    root@dell-PowerEdge-R740xd:~# ls -hl 123.txt  
    -rw-r--r-- 1 root root 0 11月 18 10:**57** 123.txt  
    root@dell-PowerEdge-R740xd:~# touch 123.txt  #更新时间  
    root@dell-PowerEdge-R740xd:~# ls -hl 123.txt  
    -rw-r--r-- 1 root root 0 11月 18 10:**58** 123.txt
